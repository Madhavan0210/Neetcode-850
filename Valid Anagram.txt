class Solution {
public:
    bool isAnagram(string s, string t) {
        int s1=s.size();
        int s2=t.size();
        if(s1!=s2) return false;
        unordered_map <char,int> mpp;
        for(int i=0;s[i] !='\0';i++) 
        {
            mpp[s[i]]++;
        }
        for(int i=0;t[i] !='\0';i++)
        {
            if(mpp[t[i]]>0) mpp[t[i]]--;
            else return false;
        }
        return true;
    }
};
